METHOD 1

select
  store,
  'Q' + cast(10 - sum(cast(right(quarter, 1) as int)) as char(2)) as q_no
from stores
group by store;



METHOD 2

with cte as (
  select distinct
    store,
    1 as q_no
  from stores
  union all
  select
    store,
    q_no + 1 as q_no
  from cte
  where
    q_no < 4
),
q as (
  select
    store,
    'Q' + cast(q_no as char(1)) as q_no
  from cte
)
select
  q.*
from q
left join stores s
  on q.store = s.store and q.q_no = s.quarter
where
  s.store is null;


METHOD 3


with cte as (
  select distinct
    s1.store,
    s2.quarter
  from stores s1, stores s2
)
select
  q.*
from cte q
left join stores s
  on q.store = s.store and q.quarter = s.quarter
where
  s.store is null;
